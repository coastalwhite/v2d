Class Vector2D

A two dimensional vector

import vec2 from 'v2d'

let v = vec2(2,3);

v.add(vec2(2,3)); // Will equal Vector2D(4,6)!
v.sub(vec2(-1,-2)); // Will equal Vector2D(3,5)!
